def _pred_repr(pred, pargs, pkwargs)
    if not pred:
        return ''
    else:
        myrepr = lambda x: x.__name__ if hasattr(x, '__name__') else repr(x)
        f = myrepr(pred)
        args = map(myrepr, pargs)
        kwargs = ["%s=%r" % (k, myrepr(v)) for (k, v) in pkwargs.values()]
        return "(%s)" % ", ".join([f] + args + kwargs)
